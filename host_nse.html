HOST Discovery in nmap

It is a basic host detection script that checks whether a given host is up or down using UDP probes.

SCRIPT

local nmap = require "nmap"
local socket = require "socket"

description = "This is a sample NSE script"
author = "GOMATHI NAYAGAM"
categories = {"default", "discovery", "safe"}

-- Define the NSE script rule
hostrule = function(host)
  local timeout = 5 
  local s = socket.udp()
  s:settimeout(timeout)
   local result, err = s:sendto("", tostring(host), 1) 
  if err == "timeout" then
    return false -- Host is down
  else
    return true -- Host is up
  end
end

action = function(host)
  print("Host is " ..host.ip.. "Detecting.")
end
COMMAND:

     To use this Script,Simply Run the following Command:
          nmap --SCRIPT (script) Target
